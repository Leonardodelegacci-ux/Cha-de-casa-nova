    <script>
        function selectValue(amount, element) {
            // Remove seleção anterior
            document.querySelectorAll('.value-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Adiciona seleção ao card clicado
            element.classList.add('selected');
            
            // Mostra seção do QR Code
            const qrSection = document.getElementById('qrSection');
            qrSection.classList.add('active');
            
            // Atualiza valores exibidos
            document.getElementById('selectedAmountDisplay').textContent = `Valor selecionado: R$ ${amount},00`;
            document.getElementById('confirmAmount').textContent = `R$ ${amount},00`;
            
            // Scroll suave até o QR Code
            qrSection.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        // Verifica se a imagem do QR Code carregou corretamente
        document.getElementById('qrImage').onerror = function() {
            this.style.display = 'none';
            this.parentElement.innerHTML += '<p style="color: #e74c3c; margin-top: 10px;">⚠️ Imagem do QR Code não encontrada.<br>Salve a imagem do QR Code como "qrcode.jpg" na mesma pasta deste arquivo.</p>';
        };
    </script>
